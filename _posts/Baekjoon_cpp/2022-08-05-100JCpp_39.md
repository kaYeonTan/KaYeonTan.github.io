```c++
#include <iostream>
#include <string>
using namespace std;

int main()
{
    int a, b, c, result;
    int count[10] = {};
    scanf("%d\n%d\n%d", &a, &b, &c);
    result = a * b * c;
    string data = to_string(result);

    for (char ch : data) count[ch - '0']++;
    for (int v : count) printf("%d\n", v);

    return 0;
}
```  
